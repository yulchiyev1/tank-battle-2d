673void RenderManager::BuildRenderMap(const std::vector<Object*>& source, Camera2D* camera)
674{
675 renderCamera = camera;
676
677 for (auto* obj : source)
678 {
679 if (!obj || !obj->IsVisible())
680 continue;
681
682 std::shared_ptr<Material> materialSP = obj->GetMaterial();
683 std::shared_ptr<Mesh> meshSP = obj->GetMesh();
684
685 SpriteAnimator* spriteAnimator = obj->GetSpriteAnimator();
686 std::shared_ptr<SpriteSheet> spriteSheetSP = spriteAnimator ? spriteAnimator->GetSpriteSheet() : nullptr;
687
688 Material* material = materialSP.get();
689 Mesh* mesh = meshSP.get();
690 SpriteSheet* spriteSheet = spriteSheetSP.get();
691 Shader* shader = material ? material->GetShader().get() : nullptr;
692
693 if (!material || !mesh || !shader)
694 continue;
695
696 uint8_t layer = renderLayerManager.GetLayerID(obj->GetRenderLayerTag()).value_or(0);
697 if (layer >= RenderLayerManager::MAX_LAYERS)
698 {
699 JIN_WRN("render skipped - invalid layer\n");
700 continue;
701 }
702
703 const float depth = obj->GetTransform2D().GetDepth();
704 const int zbin = QuantizeDepth(depth);
705
706 InstanceBatchKey key{ mesh, material, spriteSheet };
707 auto& depthMap = renderMap[layer];
708 auto& shaderMap = depthMap[zbin];
709 shaderMap[shader][key].emplace_back(obj);
710 }
711}

719void RenderManager::RegisterShader(const std::string& tag, const std::vector<std::pair<ShaderStage, FilePath>>& sources)
720{
721 if (shaderMap.find(tag) != shaderMap.end())
722 {
723 JIN_LOG("Shader with tag \"" << tag << "\" already registered.");
724 return;
725 }
726
727 auto shader = std::make_shared<Shader>();
728
729 for (const auto& [stage, path] : sources)
730 {
731 if (!shader->AttachFromFile(stage, path))
732 {
733 JIN_ERR("Failed to register shader [" << tag << "].");
734 return;
735 }
736 }
737
738 if (!shader->Link())
739 {
740 JIN_ERR("Failed to register shader [" << tag << "].");
741 return;
742 }
743
744 shaderMap[tag] = shader;
745}
