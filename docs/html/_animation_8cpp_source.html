102void SpriteAnimator::Update(float dt)
103{
104 float speedMultiplier = 1.0f;
105
106 if (playingClip && !currentClipName.empty())
107 {
108 speedMultiplier = GetClipPlaybackSpeed(currentClipName);
109 }
110
111 elapsed += dt * speedMultiplier * playbackSpeed;
112
113 if (playingClip)
114 {
115 const float dur = playingClip->frameDuration > 0.f ? playingClip->frameDuration : 0.0001f;
116
117 while (elapsed >= dur)
118 {
119 elapsed -= dur;
120 ++clipFrameIndex;
121
122 if (clipFrameIndex >= static_cast<int>(playingClip->frameIndices.size()))
123 {
124 if (playingClip->looping)
125 {
126 clipFrameIndex = 0;
127 }
128 else
129 {
130 isClipFinished = true;
131 clipFrameIndex = static_cast<int>(playingClip->frameIndices.size()) - 1;
132 currentFrame = playingClip->frameIndices[clipFrameIndex];
133 break;
134 }
135 }
136
137 currentFrame = playingClip->frameIndices[clipFrameIndex];
138 TriggerAllCallbacksForCurrentFrame();
139 }
140 }
141 else
142 {
143 const float dur = frameTime > 0.f ? frameTime : 0.0001f;
144
145 while (elapsed >= dur)
146 {
147 elapsed -= dur;
148 ++currentFrame;
149
